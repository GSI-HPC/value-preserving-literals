281 template <__arithmetic _Up>
282 consteval
283 operator _Up() const
284 {
285 using L = numeric_limits<_Up>;
286 if constexpr (floating_point<_Up>)
287 {
288 if (static_cast<unsigned long long>(static_cast<_Up>(_M_value)) != _M_value)
289 throw bad_value_preserving_cast();
290 _Up r = static_cast<_Up>(_M_value);
291 return _M_negative ? -r : r;
292 }
293 else
294 {
295 if ((_M_negative && _M_value > -static_cast<unsigned long long>(L::lowest()))
296 || (!_M_negative && _M_value > L::max()))
297 throw bad_value_preserving_cast();
298 return static_cast<_Up>(_M_negative ? -_M_value : _M_value);
299 }
300 }
301 };
